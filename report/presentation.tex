        \end{column}
    \end{columns}
    \vspace{0.8em}
    \begin{itemize}
        \item \textbf{Raw Time-Domain}: Acoustic pressure is heavily masked by ocean noise and shrimp clicks. Target motor tonals are visually indistinguishable.
        \item \textbf{LOFAR Spectrogram}: Converts signal to time-frequency, exposing stable vessel engine signatures as narrowband horizontal tonals.
    \end{itemize}
\end{frame}

\begin{frame}{Physical Challenges in Sonar Tracking}
    \begin{columns}
        \begin{column}{0.45\textwidth}
            \centering
            \includegraphics[width=\textwidth]{report/images/lofar_annotated.png}
            \captionof{figure}{Acoustic multipath harmonics.}
        \end{column}
        \begin{column}{0.55\textwidth}
            \begin{block}{Channel Obstacles to Tracking}
                \begin{itemize}
                    \item \textbf{Doppler Frequency Drift}: Vessel acceleration shifts observed frequencies:
                        $$f_{\text{obs}} = f_0 \left( \frac{c \pm v_r}{c \mp v_s} \right)$$
                        Tonal tracks ``slide'' across frequency bins.
                    \item \textbf{Multipath Echoes}: Sea floor and surface reflections create parallel phantom tracks and harmonics.
                    \item \textbf{Signal Fading}: Destructive interference causes Doppler nulls (temporary signal dropouts).
                \end{itemize}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Non-negative Matrix Factorization (NMF)}
    \begin{columns}
        \begin{column}{0.55\textwidth}
            \begin{block}{Unsupervised Source Separation}
                \begin{itemize}
                    \item Decomposes spectrogram $V \approx H \cdot W$ into static spectral components ($H$) and activation weights ($W$).
                    \item \textbf{Application}: Isolates clean narrowband engine tonals (dictionary columns) while filtering out random broadband noise.
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.45\textwidth}
            \centering
            \includegraphics[width=\textwidth]{report/images/NMF_Components_Joint_Signal.png}
            \captionof{figure}{Extracted NMF components.}
        \end{column}
    \end{columns}
\end{frame}

\section{Project Overview \& Motivation}

\begin{frame}{Project Overview \& Motivation}
    \begin{block}{Motivation}
        \begin{itemize}
            \item Traditional heuristic trackers (nearest-neighbor Kalman filters) fail when frequency thresholds are too narrow (loss of track) or too wide (noise association).
            \item We formulate sonar tracking as an MDP to learn robust, adaptive association policies.
        \end{itemize}
    \end{block}
    
    \begin{block}{Core Objectives}
        \begin{enumerate}
            \item Formulate vessel tracking as a discrete-time \textbf{Markov Decision Process (MDP)}.
            \item Build an orchestrated, dynamic tracking hierarchy (DSP Orchestrator + dynamically spawned track processors).
            \item Evaluate tabular Double Q-Learning, Linear Function Approximation (Tile Coding), and Actor-Critic policies.
            \item Establish absolute timeline matching to real-world clock time (HH:MM:SS).
        \end{enumerate}
    \end{block}
\end{frame}

\begin{frame}{Experimental Setup}
    \begin{itemize}
        \item \textbf{Datasets}: Croatia Ocean Sonics (Silba Island, Suex VRX DPV, surface motorboats) and Haifa (Seacraft GO! DVP).
        \item \textbf{Hardware}: \texttt{icListen HF ALTA} digital hydrophone sampling at 128~kHz.
        \item \textbf{Deployment}: Seabed-anchored underwater frames at 20m depth to eliminate wave and buoy motion interference.
    \end{itemize}

\section{Problem Formulation}

\begin{frame}{MDP State Space ($S$)}
    \begin{block}{Continuous State Vector}
        $$\mathbf{s}_{\text{continuous}} = (d_{\text{Hz}}, A, T, \text{age})$$
        \begin{itemize}
            \item $d_{\text{Hz}}$: Spectral distance to the mean frequency of the closest active tracker.
            \item $A$: Relative amplitude (activation weight) of the spectral peak.
            \item $T$: \textbf{Tonality score} ($1.0 - \text{Spectral Flatness}$). Measures if sound is mechanical.
            \item $\text{age}$: Continuous track age of the closest active track.
        \end{itemize}
    \end{block}

    \begin{block}{Discretized State Bins (for Tabular Agents)}
        \begin{itemize}
            \item \textbf{Distance} (6 bins): $\le 10, 25, 45, 70, 100\text{ Hz}$, out of range.
            \item \textbf{Amplitude} (5 bins): $<0.003$ to $\ge 0.03$.
            \item \textbf{Tonality} (5 bins): $<0.35$ to $\ge 0.80$.
            \item \textbf{Track Age} (4 bins): $<2.0\text{s}$ to $\ge 30.0\text{s}$.
            \item \textbf{Total State Space}: $6 \times 5 \times 5 \times 4 = 600$ discrete states.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Markov Decision Process: Action Space}
    \begin{block}{Action Set $A$ for Local Routing}
        For each candidate peak frequency, the RL agent chooses a local action:
        \begin{enumerate}
            \item \textbf{REJECT ($a = 0$)}: Ignore peak. The orchestrator takes no action, letting inactive tracking states age.
            \item \textbf{ASSOCIATE ($a = 1$)}: Route the peak frequency to update the coordinates of the closest active tracker.
            \item \textbf{SPAWN ($a = 2$)}: Start a new, concurrent vessel tracker instance at the peak's frequency.
        \end{enumerate}
    \end{block}
\end{frame}

\begin{frame}{Markov Decision Process: Decoupled Reward Rules}
    \begin{block}{Critic's Dense Feedback Calculations}
        \begin{itemize}
            \item \textbf{REJECT ($a = 0$)}:
                \begin{itemize}
                    \item \textit{Correct Reject} (ignoring ambient noise): $+2.0$
                    \item \textit{False Negative Miss} (ignoring close, highly tonal peaks): $-10.0$
                \end{itemize}
            \item \textbf{ASSOCIATE ($a = 1$)}:
                \begin{itemize}
                    \item \textit{Good Association} ($d_{\text{Hz}} \le 30\text{ Hz}$): $+10.0$
                    \item \textit{Speed Stage Change} ($30 < d_{\text{Hz}} \le 65\text{ Hz}$): $+5.0$
                    \item \textit{Bad Mismatch / Target Crossing Mismatch}: $-15.0$
                    \item \textit{Invalid Association} (associating when no trackers exist): $-20.0$
                \end{itemize}
            \item \textbf{SPAWN ($a = 2$)}:
                \begin{itemize}
                    \item \textit{Duplicate Spawn} (spawning close to active track): $-10.0$
                    \item \textit{Correct Spawn} (spawning new target with high/med tonality): $+10.0$ / $+5.0$
                \end{itemize}
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Reward Magnitudes and Degenerate Policy Prevention}
    \begin{block}{Preventing the ``Lazy'' Policy}
        \begin{itemize}
            \item If the Correct Reject reward was too high, the agent would reject every single peak to safely farm points without risk of association penalties. 
            \item Solution: Reject reward is a small baseline ($+2.0$), while good association is five times higher ($+10.0$).
        \end{itemize}
    \end{block}
    \begin{block}{Preventing ``Spawn Farming''}
        \begin{itemize}
            \item If spawning a tracker yielded equal or higher reward than maintaining it, the agent would learn to constantly drop active tracks and spawn new ones.
            \item Solution: Capped moderate spawn reward at $+5.0$, reserving $+10.0$ for Good Association to incentivize track continuity.
        \end{itemize}
    \end{block}
\end{frame}

\section{Evaluated Algorithms}

\begin{frame}{Evaluated Reinforcement Learning Algorithms}
    \begin{block}{1. Double Q-Learning (Tabular)}
        \begin{itemize}
            \item \textbf{Selection}: A simple, computationally light baseline to validate the discrete state space; resolves maximization bias using two independent Q-tables.
            \item \textbf{Behavior}: Converges quickly but suffers from track-stitching gaps due to discretization limits.
        \end{itemize}
    \end{block}
    \begin{block}{2. Linear Function Approximation (Linear FA)}
        \begin{itemize}
            \item \textbf{Selection}: To handle continuous coordinate variables directly, mapping states using high-speed overlapping \textbf{Tile Coding} (5,184 dimensions; 4 tilings of $6^4$ tiles).
            \item \textbf{Behavior}: Performed the best, achieving the highest cumulative rewards because it represents coordinate metrics continuously.
        \end{itemize}
    \end{block}
    \begin{block}{3. Actor-Critic (Stochastic Policy)}
        \begin{itemize}
            \item \textbf{Selection}: Evaluated policy gradient methods on continuous features to support probabilistic associations under fading conditions.
            \item \textbf{Behavior}: Suffered from early preference saturation due to large reward bounds. Bebehaved stably only after introducing reward scaling, gradient clipping, and preference centering.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Timeline and Normalization Discussion}
    \begin{block}{Vessel Trajectory timelines (HH:MM:SS)}
        Double Q-Learning demonstrates clean track stitching and matches vessel passing events (e.g. Ferry and support boats) perfectly to absolute time.
    \end{block}
    
    \begin{block}{Actor-Critic Normalization}
        Standard AC flatlined early due to \textbf{Softmax preference saturation} from large rewards. Resolved by:
        \begin{itemize}
            \item \textbf{Reward Scaling}: Dividing raw rewards by 20.0 (range $[-1.0, 0.5]$).
            \item \textbf{TD-Error Clipping}: Restricting gradient updates to $[-1.0, 1.0]$.
            \item \textbf{Preference Centering}: Subtracting the mean preference at each update step to prevent weights from drifting to infinity.
        \end{itemize}
    \end{block}
\end{frame}

\section{Conclusion}

\begin{frame}{Conclusion}
    \begin{block}{Summary}
        \begin{itemize}
            \item Formulated and validated an autonomous, self-supervised RL tracking framework.
            \item Linear FA achieves the highest cumulative rewards due to continuous coordinate representation.
            \item Decoupling MDP variables allows hot-swapping algorithms under a unified interface.
            \item Trained models for all evaluated architectures are available in the repository under \texttt{models/}.
        \end{itemize}
    \end{block}
    
    \begin{block}{Future Outlook}
        \begin{itemize}
            \item Further optimize continuous feature coding to reduce feature overlap crosstalk.
            \item Benchmark under severe jamming / active sonar interference scenarios.
        \end{itemize}
    \end{block}
\end{frame}
